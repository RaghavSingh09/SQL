SELECT TABLE_NAME,
CASE WHEN COUNT(TABLE_NAME)-5 <0 THEN CAST(COUNT(TABLE_NAME)-1 AS VARCHAR)+' Add Tracking Columns' ELSE CAST(COUNT(TABLE_NAME)-5 AS VARCHAR) END AS COLUMN_COUNT 
FROM INFORMATION_SCHEMA.COLUMNS A
INNER JOIN GPM_Layout_Tag B On A.TABLE_NAME = B.TG_Table_Name
WHERE B.TG_Table_Name NOT IN(SELECT Edit_Tag_Table_Name FROM GPM_Master_Tags_Edit_Details) 
GROUP BY TABLE_NAME
HAVING (COUNT(TABLE_NAME)-5 = 2 OR COUNT(TABLE_NAME)-1 =2)
